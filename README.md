# Requirements
 * git
 * python3.9
 * poetry

# Installation
 ```bash
 # install poetry
 curl -sSL https://install.python-poetry.org | python3 -

 # download codes
 git clone https://github.com/dlont/barcode.git
 ```
# Usage

 ```bash
 cd barcode
 poetry run python bc.py --help
 poetry run python bc.py "Hello, Inna" my_barcode
 ```
